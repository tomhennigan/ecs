% A4 paper with the nice article fonts etc.
\documentclass[12pt,a4paper]{article}

% Wide left margin for the binding.
% "The left hand, binding margin should not be less than 35mm. The top, right and bottom margins should not be less than 24mm."
\usepackage[left=35mm, right=24mm, top=24mm, bottom=24mm]{geometry}

% "Each paragraph should be non-indented and be separated by an empty line."
\usepackage{parskip} % http://www.tex.ac.uk/cgi-bin/texfaq2html?label=parskip
\setlength{\parindent}{0cm}
\setlength{\parskip}{0.4cm}

\usepackage{listings}
\lstset{basicstyle=\scriptsize}

% Clickable references, glossary, footnotes etc.
\usepackage[colorlinks=false,pdfborder={0 0 0}]{hyperref}

% References
\usepackage[numbers,sort&compress]{natbib}
\bibliographystyle{unsrt}
\usepackage{bibunits}

\usepackage{bibentry}
% \nobibliography*

% Make bibtex not print a heading for the references and appendix.
\renewcommand{\bibsection}

% Blue Colored URLs
\usepackage[usenames]{color}
\definecolor{urlcolor}{rgb}{0,0.08,0.45}

% You can footnote URLs with this utility command.
\usepackage{url}
\newcommand{\link}[2]{{#1}\footnote{{#1} - \footurl{#2} }}
\newcommand{\footurl}[1]{\color{urlcolor}{\underline{\url{#1}}}}

% Glossary support
\usepackage[acronym,style=list]{glossaries}
\makeglossaries
